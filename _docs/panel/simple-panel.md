---


<p class="spaced">Code-example:</p>

```js
<script src="itsabuild-min.js"></script>
<script>
    var ITSA = require('itsa'),
        modelData;

    modelData = {
        content: 'I am a panel'
    }
    document.createPanel(modelData);
    modelData.visible = true;
</script>
```
