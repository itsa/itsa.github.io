<p class="spaced">Code-example:</p>


```js
<script src="itsabuild-min.js"></script>
<script>
    var ITSA = require('itsa'),
        askName = ITSA.prompt('Please enter your name:', 'someone', 'Name');

    askName.then(function(value) {
        ITSA.alert('Your name is: '+value);
    });
</script>
```
