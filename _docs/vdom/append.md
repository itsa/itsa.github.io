```js
<script src="itsabuild-min.js"></script>
<script>
    var ITSA = require('itsa'),
        container = document.getElement('#target-container'),
        addContent;

    addContent = function(data) {
        container.append('<br>Some <i>new data</i>');
    };

    ITSA.Event.after(
        'click',
        addContent,
        '#button-get'
    );
</script>
```
